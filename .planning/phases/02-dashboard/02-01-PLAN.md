  <name>Task 1: Create telemetry Express route and register it</name>
  <files>server/routes/telemetry.js, server/index.js</files>
  <action>
Create `server/routes/telemetry.js` as a factory function matching the profiles.js pattern exactly.

CRITICAL: Use `express.Router({ mergeParams: true })` — without mergeParams, `req.params.id` is undefined when the router is mounted under `/api/profiles/:id/telemetry`.

The route queries ALL telemetry rows for the past 7 days with no aggregation — the locked decision is to send raw data and let Recharts render it. Return {recorded_at, correction_angle} only (no extra columns).

```javascript
// server/routes/telemetry.js
const express = require('express');
const router = express.Router({ mergeParams: true });

module.exports = (db) => {
  router.get('/', (req, res) => {
    const { id } = req.params;
    const rows = db.prepare(`
      SELECT recorded_at, correction_angle
      FROM telemetry
      WHERE profile_id = ?
        AND recorded_at >= datetime('now', '-7 days')
      ORDER BY recorded_at ASC
    `).all(id);
    res.json(rows);
  });
  return router;
};
```

Then in `server/index.js`, add the telemetry route registration after the existing profiles route:
```javascript
app.use('/api/profiles/:id/telemetry', require('./routes/telemetry')(db));
```
